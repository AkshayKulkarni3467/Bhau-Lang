Welcome to BhauLang

BhauLang is a Marathi-English based hybrid programming language.

•

Complied but dynamic? : Designed to be a compiled programming language with dynamic typing.

•

Keyword agnostic : Takes in a custom json file defining keywords and parses it in runtime.

•

Heavily optimized: Multiple passes during compilation for proper constant folding.


•

Requirements : GCC and NASM.

•

How to use? : ./application/blc [-asm] <input_file> [--keywords=<keyword_json>]

Add alias in bash or zsh :


alias blc = "/path_to_bhaulang/application/blc"

Commands Description :

blc <input_file>

generates an ELF-64 executable


blc -asm <input_file>

generates a parsed nasm file for <input_file>


blc <input_file> --keywords=<keywords.json>

generates and executable using custom keywords. The json format is mentioned in `Custom-Keywords` section


•

Only supports linux (x86) architecture currently.

•

Currently not implemented : postfix operators, function in function calls.

BhauLang Syntax


Assignment

bhau hai ahe a = 12;

If Statement


bhau jar(a < 10) {
  bhau parat de "small";
}

Else If Statement


bhau jar(a < 10) {
  bhau parat de "small";
}
bhau nahitar(a < 20) {
  bhau parat de "medium";
}

Else Statement


bhau jar(a < 10) {
  bhau parat de "small";
}
bhau tar {
  bhau parat de "large";
}

While Loop


bhau joparent(a < 5) {
  a = a + 1;
}

Function Definition


bhau lakshat thev fakt() {
  bhau parat de 42;
}

Return Statement


bhau parat de "result";

Entry Point


hi bhau

Exit Point


bye bhau

Switch Statement


bhau chunav(a) {
  bhau nivad 1: 
    bhau thamb;

  bhau rahudet:
    bhau thamb;
}

Case Statement


bhau nivad 2 :
  bhau parat de "two";

Break Statement


bhau thamb;

Continue Statement


bhau chalu rha;

Default Case


bhau rahudet:
  bhau thamb;

Boolean True


bhau hai ahe flag = khara;

Boolean False


bhau hai ahe flag = khota;

External Function


bhau baherun ghe printf <pseudo_name>;

Main Function


bhau ithun chalu() {
  bhau hai ahe x = 12;
}

Pointer Declaration


bhau hai ahe a = ptr b;

Reference Declaration


bhau hai ahe a = ref b;

Examples

-> control_flow.bl


hi bhau

bhau baherun ghe printf bolo;

bhau lakshat thev deep_nesting(){
  bhau hai ahe i = 1;
  bhau joparent(i <= 3){
      bhau jar(i % 2 == 0){
          bhau hai ahe j = i;
          bhau joparent(j > 0){
              bhau chunav(j){
                  bhau nivad 1:
                      bolo("From deep_nesting, choice = 1, when i = %d, j = %d\n",i,j);
                      bhau thamb;
                  bhau nivad 2:
                      bolo("From deep_nesting, choice = 2, when i = %d,j = %d\n",i,j);
                      bhau thamb;
                  bhau rahudet:
                      bolo("From deep_nesting, choice = default, when i = %d\n",i);
                      bhau thamb;
              }
              --j;
          }
      }bhau tar{
              bolo("From deep_nesting, else block when i = %d\n",i);
          }
      ++i;
  }
}

bhau ithun chalu{
  deep_nesting();
}

bye bhau

-> mixed_literals.bl


hi bhau

bhau baherun ghe printf oraoraoroara;

bhau lakshat thev mixed_conds(){
  bhau hai ahe a = 10;
  bhau hai ahe b = 2.5;
  bhau hai ahe c = 'A';
  bhau hai ahe flag = khara;
  bhau hai ahe str1 = "hello";
  bhau hai ahe str2 = "hello";

  bhau jar(a + b == 12.5 && c == 'A' && flag == 1 && str1 == str2){
      oraoraoroara("From mixed_conds , value : %d\n",1);
  }bhau tar{
      oraoraoroara("From mixed_conds , Value : %d\n",0);
  }
}

bhau ithun chalu{
  mixed_conds();
}

bye bhau

-> recursion.bl


hi bhau

bhau baherun ghe printf print;

bhau lakshat thev factorial(a){
    bhau jar(a <= 1){
        bhau parat de 1;
    }
    bhau parat de a * factorial(a - 1);
}

bhau lakshat thev fibonacci(a){
    bhau jar(a <= 1){
        bhau parat de a;
    }
    bhau parat de fibonacci(a - 1) + fibonacci(a - 2);
}

bhau lakshat thev print_recursion_tests(){
  bhau hai ahe fac_var = factorial(5);
  print("Factorial : %d\n",fac_var);
  bhau hai ahe fib_var = fibonacci(5);
  print("Fibonacci : %d\n",fib_var);
}

  bhau ithun chalu {
    print_recursion_tests();
}


bye bhau

Keywords can be defined in a json file as follows: 


{
  "<custom_assign_keyword>"    : "assign",
  "<custom_if_keyword>"        : "if",
  "<custom_elseif_keyword>"    : "elseif",
  "<custom_else_keyword>"      : "else",
  "<custom_while_keyword>"     : "while",
  "<custom_func_keyword>"      : "func",
  "<custom_return_keyword>"    : "return",
  "<custom_entry_keyword>"     : "entry",
  "<custom_exit_keyword>"      : "exit",
  "<custom_switch_keyword>"    : "switch",
  "<custom_case_keyword>"      : "case",
  "<custom_break_keyword>"     : "break",
  "<custom_continue_keyword>"  : "continue",
  "<custom_default_keyword>"   : "default",
  "<custom_true_keyword>"      : "true",
  "<custom_false_keyword>"     : "false",
  "<custom_extern_keyword>"    : "extern",
  "<custom_main_keyword>"      : "main",
  "<custom_ptr_keyword>"       : "ptr",
  "<custom_ref_keyword>"       : "ref"
}

Make sure the keywords follow this regex :

[a-z|A-Z|_][a-z|A-Z|0-9|_|$]*
